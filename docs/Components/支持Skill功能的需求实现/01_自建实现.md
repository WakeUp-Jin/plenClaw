# 自建实现：Python 从零构建 Skill 系统

## 概述

完全不依赖第三方 Agent 框架，用约 200 行 Python 实现：
- Skill 元数据扫描与渐进加载
- 关键词触发检测
- System prompt 内容注入
- 工具注册（bash / read / edit）
- tool_use 主循环

**适用模型**：任何支持 tool_use 的 LLM（Anthropic、OpenAI、Gemini 等）

## 第一部分：Skill 系统

### 数据结构

```python
import os
import re
import yaml
import subprocess
from pathlib import Path
from dataclasses import dataclass, field
from typing import Optional

@dataclass
class Skill:
    name: str
    description: str
    version: str
    dir: Path
    _body: Optional[str] = field(default=None, repr=False)

    @property
    def scripts(self) -> list[Path]:
        """返回 scripts/ 目录下的所有脚本路径"""
        scripts_dir = self.dir / "scripts"
        if scripts_dir.exists():
            return sorted(scripts_dir.iterdir())
        return []

    @property
    def references(self) -> list[Path]:
        """返回 references/ 目录下的所有文档路径"""
        refs_dir = self.dir / "references"
        if refs_dir.exists():
            return sorted(refs_dir.iterdir())
        return []
```

### 阶段一：扫描元数据（启动时执行，轻量）

```python
def parse_frontmatter(content: str) -> tuple[dict, str]:
    """解析 SKILL.md 的 YAML frontmatter，返回 (meta, body)"""
    match = re.match(r"^---\n(.*?)\n---\n?(.*)", content, re.DOTALL)
    if not match:
        return {}, content
    meta = yaml.safe_load(match.group(1)) or {}
    body = match.group(2).strip()
    return meta, body


def load_skills(skills_dir: str | Path) -> list[Skill]:
    """
    阶段一：扫描 skills/ 目录，只加载元数据（name, description）。
    正文不加载，节省 token。
    """
    skills = []
    skills_path = Path(skills_dir)

    if not skills_path.exists():
        return skills

    for skill_dir in sorted(skills_path.iterdir()):
        if not skill_dir.is_dir():
            continue

        skill_md = skill_dir / "SKILL.md"
        if not skill_md.exists():
            continue

        content = skill_md.read_text(encoding="utf-8")
        meta, _ = parse_frontmatter(content)

        if not meta.get("name") or not meta.get("description"):
            print(f"[警告] {skill_dir.name}/SKILL.md 缺少 name 或 description，跳过")
            continue

        skills.append(Skill(
            name=meta["name"],
            description=meta.get("description", ""),
            version=meta.get("version", "0.1.0"),
            dir=skill_dir,
        ))

    print(f"[Skill] 已加载 {len(skills)} 个 Skill 元数据")
    return skills
```

### 阶段二：按需加载正文（触发时执行）

```python
def load_skill_body(skill: Skill) -> str:
    """
    阶段二：Skill 被触发时，才加载 SKILL.md 正文。
    使用缓存避免重复读取。
    """
    if skill._body is not None:
        return skill._body

    skill_md = skill.dir / "SKILL.md"
    content = skill_md.read_text(encoding="utf-8")
    _, body = parse_frontmatter(content)
    skill._body = body
    print(f"[Skill] 已加载正文：{skill.name}")
    return skill._body
```

### 触发检测

```python
def detect_triggered_skills(
    skills: list[Skill],
    user_input: str,
    # 可选：更高级的向量匹配函数
    similarity_fn=None,
) -> list[Skill]:
    """
    从 description 字段中提取引号内的短语，与用户输入做关键词匹配。

    高级实现：替换 similarity_fn 为 embedding 相似度函数。
    """
    triggered = []
    user_lower = user_input.lower()

    for skill in skills:
        phrases = re.findall(r'"([^"]+)"', skill.description)
        if not phrases:
            # 没有引号短语时，尝试整体描述匹配
            phrases = skill.description.split(".")[:2]  # 取前两句

        if similarity_fn:
            # 向量相似度模式
            score = similarity_fn(user_input, " ".join(phrases))
            if score > 0.75:
                triggered.append(skill)
        else:
            # 关键词匹配模式（默认）
            if any(p.lower() in user_lower for p in phrases):
                triggered.append(skill)

    if triggered:
        names = [s.name for s in triggered]
        print(f"[Skill] 触发：{names}")

    return triggered
```

### 构建注入内容

```python
def build_skill_context(triggered: list[Skill]) -> str:
    """
    将触发的 Skill 内容构建为可注入 system_prompt 的文本。
    包含：正文内容 + 可用脚本路径列表。
    """
    if not triggered:
        return ""

    parts = ["\n\n---\n## 已加载的 Skills\n"]

    for skill in triggered:
        body = load_skill_body(skill)
        parts.append(f"\n### Skill: {skill.name} (v{skill.version})\n\n{body}")

        # 告知 Agent 可用脚本的路径（Agent 用 bash 工具执行）
        if skill.scripts:
            parts.append("\n**可用脚本（使用 bash 工具执行）：**")
            for script in skill.scripts:
                parts.append(f"- `{script.name}`: `bash {script}`")

        # 告知 references 路径（Agent 用 read_file 工具按需查看）
        if skill.references:
            parts.append("\n**参考文档（使用 read_file 工具查看）：**")
            for ref in skill.references:
                parts.append(f"- `{ref.name}`: `{ref}`")

    return "\n".join(parts)
```

```python
# ─── 工具定义（注册给 LLM 的 JSON schema）─────────────

TOOL_DEFINITIONS = [
    {
        "name": "bash",
        "description": (
            "Execute a bash command or shell script. "
            "Use this to run skill scripts, system commands, or any executable. "
            "Returns stdout and stderr combined."
        ),
        "input_schema": {
            "type": "object",
            "properties": {
                "command": {
                    "type": "string",
                    "description": "The bash command or script path to execute"
                },
                "timeout": {
                    "type": "integer",
                    "description": "Timeout in seconds (default: 30)",
                    "default": 30
                }
            },
            "required": ["command"]
        }
    },
    {
        "name": "read_file",
        "description": "Read the full contents of a file. Use for source code, configs, or skill references.",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "Absolute or relative file path"}
            },
            "required": ["path"]
        }
    },
    {
        "name": "write_file",
        "description": "Write content to a file. Creates the file if it doesn't exist.",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {"type": "string"},
                "content": {"type": "string"}
            },
            "required": ["path", "content"]
        }
    },
    {
        "name": "edit_file",
        "description": "Replace a specific string in a file with new content (exact match required).",
        "input_schema": {
            "type": "object",
            "properties": {
                "path": {"type": "string"},
                "old_string": {"type": "string", "description": "The exact text to find and replace"},
                "new_string": {"type": "string", "description": "The replacement text"}
            },
            "required": ["path", "old_string", "new_string"]
        }
    }
]

# ─── 工具执行器 ──────────────────────────────────────

def execute_tool(name: str, inputs: dict) -> str:
    """根据工具名分发执行，返回字符串结果给 LLM"""

    if name == "bash":
        return _tool_bash(inputs["command"], inputs.get("timeout", 30))

    elif name == "read_file":
        return _tool_read_file(inputs["path"])

    elif name == "write_file":
        return _tool_write_file(inputs["path"], inputs["content"])

    elif name == "edit_file":
        return _tool_edit_file(inputs["path"], inputs["old_string"], inputs["new_string"])

    else:
        return f"[error]: 未知工具 '{name}'"


def _tool_bash(command: str, timeout: int = 30) -> str:
    """Bash 工具实现：subprocess 执行，返回 stdout + stderr"""
    try:
        result = subprocess.run(
            command,
            shell=True,
            capture_output=True,
            text=True,
            timeout=timeout,
            cwd=os.getcwd(),
            env=os.environ.copy(),
        )
        output = result.stdout
        if result.stderr:
            output += f"\n[stderr]:\n{result.stderr}"
        if result.returncode != 0:
            output += f"\n[exit code]: {result.returncode}"
        return output.strip() or "(no output)"

    except subprocess.TimeoutExpired:
        return f"[error]: 命令超时（>{timeout}s）"
    except Exception as e:
        return f"[error]: {e}"


def _tool_read_file(path: str) -> str:
    try:
        return Path(path).read_text(encoding="utf-8")
    except FileNotFoundError:
        return f"[error]: 文件不存在 '{path}'"
    except Exception as e:
        return f"[error]: {e}"


def _tool_write_file(path: str, content: str) -> str:
    try:
        p = Path(path)
        p.parent.mkdir(parents=True, exist_ok=True)
        p.write_text(content, encoding="utf-8")
        return f"已写入 {path}（{len(content)} 字符）"
    except Exception as e:
        return f"[error]: {e}"


def _tool_edit_file(path: str, old_string: str, new_string: str) -> str:
    try:
        p = Path(path)
        content = p.read_text(encoding="utf-8")
        if old_string not in content:
            return f"[error]: 未找到目标字符串，请检查是否完全匹配"
        new_content = content.replace(old_string, new_string, 1)
        p.write_text(new_content, encoding="utf-8")
        return f"已编辑 {path}"
    except Exception as e:
        return f"[error]: {e}"
```

## 第三部分：Agent 主循环

```python
import anthropic

# 支持替换为其他兼容 OpenAI 格式的客户端
client = anthropic.Anthropic()

BASE_SYSTEM_PROMPT = """You are a helpful coding agent with access to tools.

When working with skills:
- Use bash tool to execute scripts listed in skill instructions
- Use read_file tool to view reference documents when needed
- Follow skill instructions precisely

Always explain what you're doing before calling a tool."""


def run_agent(
    user_input: str,
    skills_dirs: list[str] = None,
    model: str = "claude-sonnet-4-6",
    max_iterations: int = 20,
) -> str:
    """
    Agent 主入口。

    Args:
        user_input: 用户的输入
        skills_dirs: Skill 目录列表（支持多个来源）
        model: LLM 模型 ID
        max_iterations: 防止无限循环的最大 tool_use 轮数
    """
    # ── 1. 加载并触发 Skill ──────────────────────────
    all_skills = []
    for d in (skills_dirs or ["./skills"]):
        all_skills.extend(load_skills(d))

    triggered = detect_triggered_skills(all_skills, user_input)
    skill_context = build_skill_context(triggered)

    # ── 2. 构建 System Prompt ─────────────────────────
    system_prompt = BASE_SYSTEM_PROMPT + skill_context

    # ── 3. 初始化消息 ─────────────────────────────────
    messages = [{"role": "user", "content": user_input}]

    # ── 4. tool_use 主循环 ────────────────────────────
    for iteration in range(max_iterations):
        response = client.messages.create(
            model=model,
            system=system_prompt,
            messages=messages,
            tools=TOOL_DEFINITIONS,
            max_tokens=8192,
        )

        # 4a. 正常结束
        if response.stop_reason == "end_turn":
            final_text = ""
            for block in response.content:
                if hasattr(block, "text"):
                    final_text += block.text
            return final_text

        # 4b. 工具调用
        if response.stop_reason == "tool_use":
            tool_results = []

            for block in response.content:
                if block.type != "tool_use":
                    continue

                print(f"\n[工具调用] {block.name}")
                print(f"  输入: {block.input}")

                result = execute_tool(block.name, block.input)

                print(f"  结果: {result[:300]}{'...' if len(result) > 300 else ''}")

                tool_results.append({
                    "type": "tool_result",
                    "tool_use_id": block.id,
                    "content": result,
                })

            # 将工具结果追加到消息链，继续循环
            messages.append({"role": "assistant", "content": response.content})
            messages.append({"role": "user", "content": tool_results})

        else:
            # 未预期的停止原因
            print(f"[警告] 未预期的 stop_reason: {response.stop_reason}")
            break

    return "[error]: 达到最大迭代次数"
```

### 向量触发（替换关键词匹配）

```python
# 使用 sentence-transformers 做语义相似度
from sentence_transformers import SentenceTransformer, util

model = SentenceTransformer("all-MiniLM-L6-v2")

def embedding_similarity(query: str, text: str) -> float:
    emb1 = model.encode(query, convert_to_tensor=True)
    emb2 = model.encode(text, convert_to_tensor=True)
    return float(util.cos_sim(emb1, emb2))

# 传入 detect_triggered_skills
triggered = detect_triggered_skills(
    skills, user_input,
    similarity_fn=embedding_similarity
)
```
